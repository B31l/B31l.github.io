```python
def bubble_sort(a: MutableSequence) -> None:
    
    n = len(a)
    for i in range(n-1):
        for j in range(n-1, i, -1):
            if a[j-1] > a[j]:
                a[j-1], a[j] = a[j], a[j-1]
```

| O(n2) | O(n2) | O(n2) |

```python
def selection_sort(a: MutableSequence) -> None:
    
    n = len(a)
    for i in range(n - 1):
        min = i
        for j in range(i + 1, n):
            if a[j] < a[min]: min = j
        a[i], a[min] = a[min], a[i]
```
